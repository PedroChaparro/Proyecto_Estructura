package listaSimple;
                
                                public class ListNode {
                                
                                    //Elementos de cada nodo
                                
                                    private Object object;      //Objeto
                                    public ListNode next;       //Referencia (puntero) al siguiente nodo 
                                    
                                    //Constructores del objeto:
                                
                                        //Totalmente nulo
                                
                                    public ListNode() {
                                        this.object = null;
                                        this.next = null;
                                    }
                                
                                        //Entregando lo que se va a almacenar
                                
                                    public ListNode(Object object) {
                                        this.object = object;
                                        this.next = null; 
                                    }
                                
                                        //Entregando lo que se va a almacenar y a quién se va a enlazar (Siguiente nodo)
                                
                                    public ListNode(Object object, ListNode next) {
                                        this.object = object;
                                        this.next = next;
                                    }
                                
                                    //Método para enlazar a un siguiente nodo
                                
                                    public void enlazarSiguiente(ListNode next){
                                
                                        this.next = next; 
                                
                                    }
                                
                                
                                    //Mëtodo para obtener el siguiente nodo
                                
                                    public ListNode getNextTo(){
                                
                                        return this.next; 
                                
                                    }
                                
                                    //Obtener el objeto almacenado
                                
                                    public Object getObject() {
                                        return object;
                                    }
                                
                                    //Cambiar el objeto almacenado
                                
                                    public void setObject(Object object) {
                                        this.object = object;
                                    }
                                
                                    //Método para saber si dos nodos almacenan el mismo objeto 
                                    //Recibiendo como entrada el objeto a comparar
                                
                                    public boolean isEquals(Object object) {
                                        if (this.getObject().toString().equals(object.toString())) {
                                            return true;
                                        }
                                        return false;
                                    }
                                
                                    //Método para saber si dos nodos almacenan el mismo objeto
                                    //Recibiendo como entrada el nodo a comparar
                                
                                    public boolean isEquals(ListNode node) {
                                        if (this.toString().equals(node.toString())) {
                                            return true;
                                        }
                                        return false;
                                    }
                                
                                    //Convertir todos los componentes del nodo a String
                                
                                    @Override
                                    public String toString(){
                                        
                                        return "ListNode{" +
                                                "object=" + object +
                                                ", next=" + next +
                                                '}';
                                    
                                        
                                    }
                                    
                                    //Método para mostrar la información de un solo nodo
                                    
                                    public String ShowInformation(){
                                    
                                        return "ListNode{" +
                                                "object: " + object + 
                                                ", next object: " + this.next.getObject() + 
                                                "}";
                                        
                                    }
                                    
                                }

                                            } else {
                                                //Se crea la nueva cabeza
                                                ListNode newHead = new ListNode(object); 
                                                //El siguiente elemento va a ser la anterior cabeza
                                                newHead.next = head; 
                                                //La cabeza será el nuevo nodo
                                                head = newHead; 
                                            }
                                            this.size++;
                                            return true;
                                        } catch (Exception e) {
                                            return false;

                                                head = new ListNode(object);
                                                tail = head;    //Como solo hay un elemento, a la cola se le asigna el mismo valor que la cola
                                            } else {
                                
                                                //Se crea un nodo, y dentro de ese nodo se guarda el objeto 
                                                ListNode newTail = new ListNode(object);
                                                //El siguiente elemento a la anterior cola va a ser el nuevo nodo
                                                tail.next = newTail; 
                                                //El valor "cola" es tomado por el nuevo nodo
                                                tail = newTail;  
                                            }
                                
                                            //Si el elemento fue almacenado, se suma 1 al tamaño de la lista y se retorna true
                                
                                            this.size++;
                                            return true;
                                        } catch (Exception e) {
                                            return false;
                                        }
                                    }
