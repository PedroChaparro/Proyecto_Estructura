package Doble;
                                import java.util.Scanner;
                                
                                public class doublelist {
                                
                                    Nodo primero;
                                    Nodo ultimo;
                                    Scanner entrada = new Scanner(System.in);
                                
                                    public doublelist(){
                                        primero = null;
                                        ultimo = null;
                                    }
                                
                                    public void ingresarNodo(Object dato){
                                        Nodo nuevo = new Nodo();
                                        nuevo.dato = dato;
                                
                                        if(primero == null){
                                            primero = nuevo;
                                            primero.siguiente = null;
                                            primero.anterior = null;
                                            ultimo = primero;
                                        }else{
                                            ultimo.siguiente = nuevo;
                                            nuevo.anterior = ultimo;
                                            nuevo.siguiente = null;
                                            ultimo = nuevo;
                                        }
                                
                                    }
                                
                                    public void ListaAscendente(){
                                
                                        Nodo actual = new Nodo();
                                        actual = ultimo;
                                        while(actual != null){
                                            System.out.println("- "+actual.dato);
                                            actual = actual.anterior;
                                        }
                                    }
                                
                                    public void ListaDescendente(){
                                
                                        Nodo actual = new Nodo();
                                        actual = primero;
                                        while(actual != null){
                                            System.out.println("- "+actual.dato);
                                            actual = actual.siguiente;
                                        }
                                    }
                                
                                    public void BuscarNodo(Object dato){
                                        Nodo actual = new Nodo();
                                        actual = primero;
                                        while(actual != null){
                                            if(actual.dato.equals(dato)){
                                                System.out.println("Nodo encontrado: Existe");
                                                System.out.println("");
                                            }
                                            actual = actual.siguiente;
                                        }
                                    }
                                
                                    public void ModificarNodo(Object dato){
                                        Nodo actual = new Nodo();
                                        actual = ultimo;
                                        while(actual != null){
                                            if(actual.dato.equals(dato)){
                                                System.out.println("Ingrese el nuevo Nodo: ");
                                                actual.dato = entrada.next();
                                            }
                                            actual = actual.anterior;
                                        }
                                    }
                                
                                    public void EliminarNodo(Object dato){
                                        Nodo actual = new Nodo();
                                        Nodo atras = new Nodo();
                                        actual = primero;
                                        atras = null;
                                        while(actual != null){
                                            if(actual.dato.equals(dato)){
                                                if(actual.equals(primero)){
                                                    primero = primero.siguiente;
                                                    primero.anterior = null;
                                
                                                }else{
                                                    atras.siguiente = actual.siguiente;
                                                    actual.siguiente.anterior = actual.anterior;
                                                }
                                            }
                                            atras = actual;
                                            actual = actual.siguiente;
                                        }
                                
                                    }
                                
                                    
                                
                                }
