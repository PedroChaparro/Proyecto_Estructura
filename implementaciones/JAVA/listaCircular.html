    package Circular;

    public class Nodo{

        int dato;
        Nodo next;
    
        public Nodo(int d){
            dato = d;
            next = this;
        }
    }

    package Circular;
    public class ListaCircular {
    
        Nodo last;
    
        public ListaCircular(){
            last = null;
        }
    
        //Metodo para saber cuando la lista está vacia
    
        public boolean isEmpty(){
            return last == null;
        }
    
        //Metodo para insertar Nodos
    
        public ListaCircular insert(int element){
            Nodo nuevo = new Nodo(element);
            if(last != null){
                nuevo.next = last.next;
                last.next = nuevo;
            }
            last = nuevo;
            return this;
        }
    
        //Metodo para mostrar la lista
    
        public void showList(){
            Nodo aux = last.next;
            String cadena = "";
            do{  
                cadena = cadena + "{ " + aux.dato + " } -> ";   
            }while(aux != last.next);
            System.out.println("Mostrando la lista circular");
        }
    
        //Metodo para eliminar un Nodo de la lista
    
        public boolean remove(int element){
            Nodo actual;
            boolean find = false;
            actual = last;
            while(actual.next != last && !find){
                find = (actual.next.dato == element);
                if(!find){
                    actual = actual.next;
                    if(find){
                        Nodo aux = actual.next;
                        if(last == last.next){
                            last = null;
                        }else{
                            if(aux == last){
                                last = actual;
                            }
                            actual.next = aux.next;
                        }
                        aux = null;
                    }
                    return find == true;
                }
            }
            return find;
        }
    }
